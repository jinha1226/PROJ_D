# Shop & Gold System Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add gold economy (floor scatter, humanoid monster drops, orc treasure room) and a DCSS-style shop tile system with purchase dialog, partial spellbooks, and spellpages.

**Architecture:** Shop state (`_shop_items`, `_shop_is_special`) lives in `Game.gd` and is included in the floor cache. `DungeonMap` gains a `SHOP` tile. `ItemRegistry` gains a spell pool and `generate_partial_book()`. `ShopDialog` follows the `BagDialog` static-class pattern using `GameDialog`.

**Tech Stack:** GDScript 4, Godot 4.6, `.tres` resources, `GameDialog` / `UICards` UI primitives.

## Returns an item entry dict for a runtime partial spellbook (2-3 random spells).
## The entry is NOT backed by a static .tres — Player.use_item reads grants_spell_ids
## from the entry dict when present.
func generate_partial_book(depth: int) -> Dictionary:
    var pool: Array = []
    for s in SPELL_POOL:
        if int(s.get("depth_min", 1)) <= depth:
            pool.append(s)
    pool.shuffle()
    var count: int = randi_range(2, 3)
    var chosen: Array = pool.slice(0, min(count, pool.size()))
    var spell_ids: Array = []
    var max_depth_min: int = 1
    for s in chosen:
        spell_ids.append(s["id"])
        max_depth_min = max(max_depth_min, int(s.get("depth_min", 1)))
    var tier: int = clampi((max_depth_min + 1) / 2, 1, 5)
    var tile: String = _PARTIAL_BOOK_TILES[randi() % _PARTIAL_BOOK_TILES.size()]
    var name: String = _PARTIAL_BOOK_NAMES[randi() % _PARTIAL_BOOK_NAMES.size()]
    return {
        "id": "book_partial",
        "plus": 0,
        "grants_spell_ids": spell_ids,
        "display_name": name,
        "tile_path": tile,
        "tier": tier,
        "kind": "book",
    }
```

Add this helper function in `Game.gd`:

## Returns true if this depth is designated the shop floor for its 5-floor block.
## Each block has a 70% chance of having exactly one shop, placed on a random
## interior floor of the block (not the first or last).
func _is_shop_floor(depth: int) -> bool:
    var block_start: int = ((depth - 1) / 5) * 5 + 1  # 1, 6, 11, 16
    var rng := RandomNumberGenerator.new()
    rng.seed = _floor_seed(block_start) ^ 0xABCD1234
    if rng.randf() >= 0.70:
        return false  # No shop this block
    # Pick one depth inside the block (avoid first and last of block).
    var block_end: int = block_start + 4
    var shop_depth: int = rng.randi_range(block_start + 1, block_end - 1)
    return depth == shop_depth
```

const THUMB_SIZE := 48

static func open(shop_items: Array, is_special: bool, player: Player, parent: Node,
        on_purchase: Callable = Callable()) -> void:
    var title: String = LocaleManager.t("SHOP_TITLE_SPECIAL") if is_special \
            else LocaleManager.t("SHOP_TITLE")

static func _populate(dlg: GameDialog, shop_items: Array, player: Player,
        on_purchase: Callable) -> void:
    var body: VBoxContainer = dlg.body()
    if body == null:
        return
    for child in body.get_children():
        child.queue_free()
    body.add_theme_constant_override("separation", GameTheme.PAD_M)

    # Gold header
    var gold_lbl := Label.new()
    gold_lbl.text = LocaleManager.t("SHOP_GOLD") % player.gold
    gold_lbl.add_theme_font_size_override("font_size", GameTheme.TYPO_BODY_LARGE)
    gold_lbl.add_theme_color_override("font_color", Color(1.0, 0.88, 0.3))
    body.add_child(gold_lbl)

    if shop_items.is_empty():
        var empty := Label.new()
        empty.text = "Nothing for sale."
        empty.add_theme_color_override("font_color", Color(0.55, 0.55, 0.6))
        body.add_child(empty)
        return

    for i in shop_items.size():
        var slot: Dictionary = shop_items[i]
        body.add_child(_make_row(slot, i, player, dlg, shop_items, on_purchase))

static func _make_row(slot: Dictionary, idx: int, player: Player, dlg: GameDialog,
        shop_items: Array, on_purchase: Callable) -> HBoxContainer:
    var row := HBoxContainer.new()
    row.add_theme_constant_override("separation", GameTheme.PAD_S)

    var entry: Dictionary = slot.get("entry", {})
    var item_id: String = String(entry.get("id", ""))
    var base_id: String = ItemRegistry.base_id_of(item_id) if ItemRegistry != null else item_id
    var data: ItemData = ItemRegistry.get_by_id(base_id) if ItemRegistry != null else null
    var price: int = int(slot.get("price", 0))
    var sold: bool = bool(slot.get("sold", false))
    var is_known: bool = GameManager.is_identified(base_id) if GameManager != null else false

    # Thumbnail
    var tex_path: String = entry.get("tile_path", data.tile_path if data != null else "")
    if tex_path == "" and data != null:
        tex_path = data.tile_path
    var thumb := TextureRect.new()
    thumb.custom_minimum_size = Vector2(THUMB_SIZE, THUMB_SIZE)
    thumb.stretch_mode = TextureRect.STRETCH_KEEP_ASPECT_CENTERED
    if tex_path != "":
        thumb.texture = load(tex_path) as Texture2D
    row.add_child(thumb)

    # Name
    var name_lbl := Label.new()
    var display_name: String = entry.get("display_name",
            ItemRegistry.entry_display_name(entry) if ItemRegistry != null else item_id)
    name_lbl.text = display_name
    name_lbl.size_flags_horizontal = Control.SIZE_EXPAND_FILL
    name_lbl.add_theme_font_size_override("font_size", GameTheme.TYPO_BODY_LARGE)
    # White = already identified, Blue = unidentified in player's knowledge
    if sold:
        name_lbl.add_theme_color_override("font_color", Color(0.4, 0.4, 0.4))
    elif is_known:
        name_lbl.add_theme_color_override("font_color", Color.WHITE)
    else:
        name_lbl.add_theme_color_override("font_color", Color(0.5, 0.7, 1.0))
    row.add_child(name_lbl)

    # Price / sold label
    var price_lbl := Label.new()
    price_lbl.text = LocaleManager.t("SHOP_SOLD_OUT") if sold else "%dg" % price
    price_lbl.add_theme_color_override("font_color",
            Color(0.5, 0.5, 0.5) if sold else Color(1.0, 0.88, 0.3))
    row.add_child(price_lbl)

    # Buy button
    if not sold:
        var btn := Button.new()
        btn.text = LocaleManager.t("SHOP_BUY")
        btn.disabled = player.gold < price
        btn.pressed.connect(func():
            if player.gold < price:
                CombatLog.post(LocaleManager.t("SHOP_NOT_ENOUGH_GOLD"), Color(1.0, 0.4, 0.4))
                return
            player.gold -= price
            player.emit_signal("stats_changed")
            # Identify immediately
            GameManager.identify(base_id)
            # Add to inventory (book/spellpage: study on use; others: normal item)
            var buy_entry: Dictionary = entry.duplicate(true)
            player.items.append(buy_entry)
            shop_items[idx]["sold"] = true
            if not on_purchase.is_null():
                on_purchase.call()
            # Rebuild dialog
            _populate(dlg, shop_items, player, on_purchase)
        )
        row.add_child(btn)

    return row
```

func _generate_shop_inventory(depth: int) -> void:
    _shop_items.clear()
    var count: int = randi_range(4, 6)

    if _shop_is_special:
        # Special shop: 3-4 randarts + 1-2 consumables
        for _i in range(randi_range(3, 4)):
            var eq: ItemData = ItemRegistry.pick_equipment(depth) if ItemRegistry != null else null
            if eq == null:
                continue
            var entry: Dictionary = ItemRegistry.make_entry(eq.id, depth)
            # Force randart stamp
            if not entry.has("artifact_name"):
                entry = ItemRegistry.make_entry(eq.id, depth + 5)
            _shop_items.append({"entry": entry, "price": _shop_price(eq, true), "sold": false})
        for _i in range(randi_range(1, 2)):
            var cons: ItemData = ItemRegistry.pick_kind(depth, "potion") if ItemRegistry != null else null
            if cons == null:
                continue
            _shop_items.append({"entry": {"id": cons.id, "plus": 0}, "price": _shop_price(cons, false), "sold": false})
        return

    # Normal shop: mixed items
    for _i in range(count):
        var roll: float = randf()
        var entry: Dictionary = {}
        var data: ItemData = null
        if roll < 0.22 and ItemRegistry != null:
            data = ItemRegistry.pick_kind(depth, "potion")
            if data != null: entry = {"id": data.id, "plus": 0}
        elif roll < 0.42 and ItemRegistry != null:
            data = ItemRegistry.pick_kind(depth, "scroll")
            if data != null: entry = {"id": data.id, "plus": 0}
        elif roll < 0.60 and ItemRegistry != null:
            data = ItemRegistry.pick_equipment_weighted(depth)
            if data != null: entry = ItemRegistry.make_entry(data.id, depth)
        elif roll < 0.78 and ItemRegistry != null:
            entry = ItemRegistry.generate_partial_book(depth)
            data = ItemRegistry.get_by_id("book_fire")  # dummy for pricing tier
            if entry.is_empty(): entry = {}
        else:
            # Spellpage
            data = ItemRegistry.pick_kind(depth, "spellpage") if ItemRegistry != null else null
            if data != null: entry = {"id": data.id, "plus": 0}

        if entry.is_empty() or (data == null and not entry.has("tier")):
            continue
        var is_rare: bool = entry.has("artifact_name")
        var tier: int = int(entry.get("tier", data.tier if data != null else 1))
        var price: int = _shop_price_by_tier(tier, String(entry.get("kind", data.kind if data != null else "")), is_rare)
        _shop_items.append({"entry": entry, "price": price, "sold": false})

func _shop_price_by_tier(tier: int, kind: String, is_randart: bool) -> int:
    var consumable_kinds: Array = ["potion", "scroll", "spellpage", "book"]
    var base: int
    if is_randart:
        base = [0, 0, 0, 0, 220, 320][clampi(tier, 0, 5)]
    elif consumable_kinds.has(kind):
        base = [0, 15, 25, 40, 60, 80][clampi(tier, 0, 5)]
        if kind == "book":
            base = int(base * 1.5)
    else:
        base = [0, 40, 70, 110, 160, 220][clampi(tier, 0, 5)]
    return max(base, 5)

func _shop_price(data: ItemData, is_randart: bool) -> int:
    return _shop_price_by_tier(data.tier, data.kind, is_randart)
```
